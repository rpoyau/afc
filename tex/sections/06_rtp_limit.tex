\section{Stochastic Dynamics and RTP Limit}\label{sec:rtp-limit}

\begin{definition}[Local Memorum functional]\label{def:memorum-functional}
Let a finite plan be given with flux $\tau_t$ as in Definition~\ref{def:flux-prod}, and let a directed chain (Diatia) be indexed by discrete steps $k = 0,\dots,L-1$ with realised outcomes $\tau_k \in T$. A \emph{local Memorum functional} along this chain is an integer
\[
    M := \sum_{k=0}^{L-1} \tau_k.
\]
The functional $M$ parametrises a family of step-local kernels on $T$,
\[
    S_M : T \to [0,1], \qquad \sum_{t \in T} S_M(t) = 1,
\]
which deform the uniform kernel of Consequence~\ref{cons:uniform} according to the Memorum accumulated along the chain.
\end{definition}

\begin{consequence}[Microscopic run-and-tumble dynamics]\label{cons:rt-micro}
Let a walker on the graph carry a state consisting of a node $x \in V$, a direction along an incident seat, and a local Memorum functional $M$ as in Definition~\ref{def:memorum-functional}. At each tally the walker:
\begin{itemize}
    \item samples a flux $\tau \in T$ from the kernel $S_M$;
    \item updates its Memorum $M \mapsto M + \tau$;
    \item moves along the current direction if $\tau \ne 0$ (run) or randomises its direction if $\tau = 0$ (tumble).
\end{itemize}
The resulting microscopic dynamics on the graph is a gated run-and-tumble walk driven entirely by integer outcomes in $T$.
\end{consequence}

\begin{theorem}[Diffusion / RTP limit]\label{thm:rtp-limit}
Consider the run-and-tumble dynamics of Consequence~\ref{cons:rt-micro} on a family of graphs obtained by refining a fixed domain, with seat length $\varepsilon \to 0$ and tally spacing $\Delta t \to 0$ such that $\varepsilon^2 / \Delta t$ converges to a finite diffusion scale. For $N \gg 1$ bips in this scaling regime, the empirical distribution of walker positions and directions converges, in the coarse-grained limit, to a continuous diffusion or run-and-tumble (RTP) kernel on the domain. Equivalently, the law of paths converges to a Feynman--Kac-type kernel determined by the effective drift and diffusion induced by the family $\{S_M\}$.
\end{theorem}

\begin{remark}[Counts-first origin of the continuum]\label{rem:counts-first}
The construction of Consequence~\ref{cons:rt-micro} and Theorem~\ref{thm:rtp-limit} uses only integer counts of ternary outcomes on seats and Memorum functionals built from those counts. The continuous diffusion or Feynman--Kac kernel appears as a scaling limit of the discrete process and is not assumed as a premise of the calculus.
\end{remark}

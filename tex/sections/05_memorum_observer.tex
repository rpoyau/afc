\section{Memorum and Observer Consequences}\label{sec:memorum-observer}

\begin{definition}[Observation path]\label{def:obs-path}
Let a finite plan be represented as in Consequence~\ref{cons:plan}, with node set $V$, seat set $E$, and tally $t \in \mathbb{Z}$. Fix a finite window $t = 0,\dots,T_{\max}-1$ and a finite interface $E_{\mathrm{plan}} \subseteq E$. An \emph{observation path} on this plan is the finite collection
\[
    \mathcal{T}
    :=
    \bigl\{
        \tau_t(e)
        \;\big|\;
        e \in E_{\mathrm{plan}},\;
        t = 0,\dots,T_{\max}-1
    \bigr\},
\]
where each $\tau_t(e) \in T$ is the realised ternary outcome at seat $e$ and tally $t$.
\end{definition}

\begin{definition}[Observer ledger and Memorum]\label{def:memorum}
Let an observer be equipped with a finite internal ledger state $m_t$ at each tally $t$ in a declared window $t = 0,\dots,T_{\max}-1$. The \emph{observer ledger} over the window is the map
\[
    t \mapsto m_t.
\]
The corresponding \emph{Memorum} is the finite sequence
\[
    \mathcal{M}
    :=
    \{m_t\}_{t=0}^{T_{\max}-1}.
\]
Two observers that generate the same Memorum on the same plan window are identified.
\end{definition}

\begin{consequence}[Path selection]\label{cons:path-selection}
From Axiom~A1 and A2, a single execution of a finite plan produces exactly one observation path $\mathcal{T}$ and a paired memory path $\mathcal{M}$ as in Definitions~\ref{def:obs-path} and~\ref{def:memorum}. For a given plan and window, all future memory states $m_t$ along that execution are constrained by the realised past $\{\tau_s(e)\}_{s \le t,\,e \in E_{\mathrm{plan}}}$.
\end{consequence}

\begin{consequence}[Replication of Memora inside Null]\label{cons:replication}
By A3 (Null $\leq$ Observer), for any Memorum $\mathcal{M}$ generated on a completed observation loop and any finite integer $K \ge 0$, Null admits a configuration with $K$ disjoint relational copies of $\mathcal{M}$. In particular, Null supports arbitrary finite replication of observer memory patterns arising from finite plans.
\end{consequence}

\begin{consequence}[Fractal, scale-invariant, isotropic Memorum space]\label{cons:fractal-memorum}
From A3 and A5, the collection of Memora generated by finite plans has the following properties:
\begin{itemize}
    \item \emph{Fractal}: finite Memora can be embedded as subpatterns of larger Memora and replicated at multiple scales inside Null;
    \item \emph{Scale invariant}: rescaling a plan by duplicating or grouping identical segments produces Memora related by finite concatenations and embeddings;
    \item \emph{Isotropic}: Memora related by relabellings of observer patches or plan interfaces are equivalent.
\end{itemize}
The Memorum space is thus closed under finite replication, embedding, and isotropic relabelling inside the Null Infinite Potential.
\end{consequence}

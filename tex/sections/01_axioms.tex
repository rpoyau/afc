\section{Axiom: Null Infinite Potential}\label{sec:axiom}

\begin{axiom}[A1]\label{ax:a1}
\[
    \varnothing \;\equiv\; \infty \;\equiv\; \chi \;\longrightarrow\; \xi,
    \qquad
    \langle \xi \rangle = 0
    \;\equiv\;
    \operatorname{Var}(\xi) \to \infty.
\]
Relationally zero distinction is a Null infinite potential that is stochastically chaos.
\end{axiom}


\begin{consequence}[A2: Plan $\rightarrow$ Observe]\label{cons:a2}
To observe is to become stochastically coupled to the Null potential along a finite plan, selecting a single stochastic path of outcomes for as long as the coupling holds.
\end{consequence}

\begin{consequence}[A3: Null $\leq$ Observer]\label{cons:a3}
Null admits arbitrarily many relational copies of any completed observer. Schematically,
\[
    \mathrm{Null} = K \cdot \mathrm{Observer}
\]
for any finite integer $K \ge 0$, with $K = 0$ corresponding to no completed observation loop.
\end{consequence}

\begin{consequence}[A4: Observer = Memory]\label{cons:a4}
Observer $=$ Memory: over any finite phase the total memory trace of an observer is its Memorum.
\end{consequence}

\begin{consequence}[A5: Memorum structure]\label{cons:a5}
The Memorum is fractal, scale invariant, and isotropic.
\end{consequence}
